library(RODBC)
library(tidyverse)
library(openxlsx)

start_date<-‘2019-06-01’

与mysql建立连接通道

channel<-odbcConnect(‘mysql_data’,uid = ‘‘,pwd = ‘‘)

查看库中(即通道)所有表

sqlTables(channel)

贷后数据查询

data<-sqlQuery(channel,str)

按照商户名称、原始逾期状态进行计数

show_data<-data%>%filter(as.Date(original_real_recive_time,’%Y-%m-%d’)%
select(original_real_recive_time,repay_state_tj,fy_merchant_name,fy_system_sn)%>%
group_by(fy_merchant_name,repay_state_tj)%>%count(repay_state_tj)

对商户名称中NA缺失数据，更改为字符型NA

show_data$fy_merchant_name[which(is.na(show_data$fy_merchant_name),)]<-‘NA’

对原始逾期状态为空的缺失数据，更改为字符空白

show_data$repay_state_tj[which(show_data$repay_state_tj==””,)]<-‘空白’

对show_data行列转换

show_data_trans<-tapply(show_data$n,list(show_data$fy_merchant_name,show_data$repay_state_tj),sum)
show_data_trans

贷后数据导出

创建列表，包含导出的两个sheet表

v<-list(‘贷后数据源’=data,’贷后展示’=as.data.frame(show_data_trans))

数据导出

nn<-paste(‘D:/数据质量/R输出结果/贷后数据’,Sys.Date(),’.xlsx’,sep = ‘’)
write.xlsx(v,file = nn,asTable = c(T,T),withFilter=c(T,T),colwiths=c(‘auto’,’auto’),row.names=T)

str2<-paste(“select fy_system_sn
,user_id
,user_name
,user_id_card
,phone_nm
,device_type
,product_code
,case when user_type = ‘new’ then ‘首借’
when user_type = ‘old’ then ‘复借’
else ‘未判定’ end as user_type
,channel_code
,channel_name
,strategybranch_id
,strategybranch_name
,from_unixtime(borrow_time,’%Y-%m-%d’) as borrow_time
,from_unixtime(risk_begin_time,’%Y-%m-%d’) as risk_begin_time
,from_unixtime(real_recive_time,’%Y-%m-%d’) as real_recive_time
,borrow_amount
,real_recive_amount
,case when risk_audit_result = ‘ACCEPT’ then ‘通过’
when risk_audit_result = ‘REJECT’ then ‘拒绝’
when risk_audit_result = ‘REVIEW’ then ‘复审’
else ‘’ end as risk_audit_result
,case when manual_audit_result = ‘0’ then ‘未认证’
when manual_audit_result = ‘10’ then ‘申请’
when manual_audit_result = ‘20’ then ‘通过’
when manual_audit_result = ‘30’ then ‘驳回’
when manual_audit_result = ‘40’ then ‘拒绝’
else ‘’ end as manual_audit_result
,case when risk_audit_result = ‘REJECT’ or manual_audit_result = ‘40’ then ‘拒绝’
when risk_audit_result = ‘ACCEPT’ or manual_audit_result = ‘20’ then ‘通过’
when risk_audit_result = ‘REVIEW’ and manual_audit_result = ‘0’ then ‘人工未认证’
when risk_audit_result = ‘REVIEW’ and manual_audit_result = ‘10’ then ‘申请’
when risk_audit_result = ‘REVIEW’ and manual_audit_result = ‘30’ then ‘驳回’
else ‘’ end as final_audit_result
,fy_merchant_id
,fy_merchant_name
from cl_property_apply_info
where borrow_time >= “,”‘“,start_date,”‘“,sep=’’)
